# filling the age colunm with the average age for each class
def fillin(columns):
    age = columns[0]
    pclass = columns[1]
    if pd.isnull(age):
        if pclass == 1:
            return 37
        elif pclass == 2:
            return 29
        else:
            return 24
    else: return age
